
// This is code to describe the algebra of a_2.
// It is written in the programming language Magma.

PQ<a,b,c,d,e,f,g,h,i>
:= PolynomialRing(RationalField(), 9);

a2Relns := [
a + e + i,
a*e - b*d,
a*f - c*d,
a*h - b*g,
a*i - c*g,
b*f - c*e,
b*i - c*h,
d*h - e*g,
d*i - f*g,
e*i - f*h
];

a2RelnId := ideal<PQ | a2Relns>;

Dimension(a2RelnId) eq 4;
IsPrime(a2RelnId);


// The matrix associated is
// [a,b,c]
// [d,e,f]
// [g,h,i]





// Consider a blow-up at the ideal <a,b,c>.
//
// Due to not having a bidegree to hand,
// and the relative simplicity of the \C[a_2] coordinate algebra,
// we compute this manually.


// Consider the a chart.
// This has new generators b/a and c/a.
// Clearly b and c are redundant.
// We find the following further redundancies:

a - (-e - i) in a2RelnId;
PQ ! (a* (e - d * b/a) ) in a2RelnId;
PQ ! (a* (f - d * c/a) ) in a2RelnId;
PQ ! (a* (h - g * b/a) ) in a2RelnId;
PQ ! (a* (i - g * c/a) ) in a2RelnId;

// This leaves us with only 4 generators.
// This confirms that the a chart is isomorphic to \C^4,
// and is generated by d, g, b/a and c/a,
// as stated in my thesis.


// Consider the b chart.
// This has new generators a/b and c/b.
// Clearly a and c are redundant.
// We find the following further redundancies:

a - (-e - i) in a2RelnId;
PQ ! (b* (d - e * a/b) ) in a2RelnId;
PQ ! (b* (f - e * c/b) ) in a2RelnId;
PQ ! (b* (g - h * a/b) ) in a2RelnId;
PQ ! (b* (i - h * c/b) ) in a2RelnId;

// This leaves us with only 4 generators.
// This confirms that the b chart is isomorphic to \C^4,
// and is generated by e, h, a/b and c/b,
// as stated in my thesis.


// Consider the c chart.
// This has new generators a/c and b/c.
// Clearly a and b are redundant.
// We find the following further redundancies:

a - (-e - i) in a2RelnId;
PQ ! (c* (d - f * a/c) ) in a2RelnId;
PQ ! (c* (e - f * b/c) ) in a2RelnId;
PQ ! (c* (g - i * a/c) ) in a2RelnId;
PQ ! (c* (h - i * b/c) ) in a2RelnId;

// This leaves us with only 4 generators.
// This confirms that the c chart is isomorphic to \C^4,
// and is generated by f, i, a/c and b/c,
// as stated in my thesis.
